    \part Prove that $T(n) \leq \lceil \log_{2}n\rceil$ for all natural numbers $n \geq 2$.

    \begin{customsolutionbox}
        \begin{proof}
            Let $P(n)$ be true if $T(n) \leq \lceil\log_{2}n\rceil$ and $T(n+1) \leq \lceil\log_{2}n\rceil$ where $n \geq 2$, otherwise false. \\

            Base case $n = 2$ for $T(n)$: \\
            \begin{gather}
                T(2) \leq \lceil\log_22\rceil \\
                \lceil\log_22\rceil = 1 \\
                T(2) \leq 1
            \end{gather}

            Base case $n = 2$ for $T(n+1)$
            \begin{gather}
                T(2+1) \leq \lceil\log_{2}2\rceil \\
                T(3) \leq 1
            \end{gather}
            This establishes for base case $n = 2$ \\

            Inductive case: Let's fix $k \geq 2$. Assume $P(a)$ is true for $2 \leq a \leq k$. We need to show that $P(k+1)$ holds for both recurrence equations. \\

            Now we need to prove that $T(k+1) \leq \lceil\log_{2}k+1\rceil$ and $T(k+2) \leq \lceil\log_{2}k+1\rceil$ :
            \begin{align*}
                T(k+1) &\leq \lceil\log_2k+1\rceil \\
                &\leq  \lceil\log_2k\rceil+ \log_22 \\
                T(k+2) &\leq \lceil\log_2k + 1\rceil \\
                &\leq \lceil\log_2k\rceil+ \log_22
            \end{align*}

            From the recurrence relation $T(k+1)$ we have:
            \begin{align*}
                T(k+1) &\leq T(\lceil \frac{k+2}{2}\rceil) + 1 \\
                &\leq T(\lceil \frac{k+1}{2} + \frac{1}{2}\rceil) + 1 \\
                &\leq T(\frac{k+1}{2} + 1) + 1 \hspace{5mm} \text{by inductive hypothesis } T(a+1) = \lceil\log_2k\rceil \\
                &\leq  \lceil\log_2k\rceil +1 \\
                &\leq \lceil\log_2k\rceil + \log_22
            \end{align*}


            Similar to the previous recurrence we have $T(k+2)$ as follows:
            \begin{align*}
                T(k+2) &\leq T(\lceil \frac{k+3}{2}\rceil) + 1 \\
                &\leq T(\lceil \frac{k+1}{2} + 1\rceil) + 1 \\
                &\leq T(\frac{k+1}{2} + 1) + 1 \hspace{5mm} \text{by inductive hypothesis } T(a+1) = \lceil\log_2k\rceil \\
                &\leq  \lceil\log_2k\rceil +1 \\
                &\leq \lceil\log_2k\rceil + \log_22
            \end{align*}

            Therefore, by induction we have conclude that $T(n) \leq \lceil \log_2n \rceil$ for all $ n\geq 2$
        \end{proof}
    \end{customsolutionbox}
